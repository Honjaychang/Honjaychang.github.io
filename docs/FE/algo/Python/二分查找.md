### 正常

```python
from typing import List

class Binary:
    def serach(self, nums: List[int], target: int) -> int:
        left, right = 0, len(nums) - 1
        while left <= right:
            mid = left + (right - left) // 2
            if nums[mid] < target:
                left = mid + 1
            elif nums[mid] > target:
                right = mid - 1
            elif nums[mid] == target:
                return mid
        return -1

bin = Binary()
print(bin.serach([1, 2, 2, 2, 3], 2))  # 2
```

#### 左

```python
from typing import List

class Left:
    def search(self, nums: List[int], target: int) -> int:
        left, right = 0, len(nums) - 1
        while left <= right:
            mid = left + (right - left) // 2
            if nums[mid] < target:
                left = mid + 1
            elif nums[mid] > target:
                right = mid - 1
            elif nums[mid] == target:
                right = mid - 1
        if left > len(nums) | nums[left] != target:
            return -1
        return left

le = Left()
print(le.search([1, 2, 2, 2, 3], 2))  # 1
```

#### 右

```python
from typing import List

class Right:
    def search(self, nums: List[int], target: int) -> int:
        left, right = 0, len(nums) - 1
        while left <= right:
            mid = left + (right - left) // 2
            if nums[mid] < target:
                left = mid + 1
            elif nums[mid] > target:
                right = mid - 1
            elif nums[mid] == target:
                left = mid + 1
        if right < 0 | nums[right] != target:
            return -1
        return right

right = Right()
print(right.search([1, 2, 2, 2, 3], 2))  # 3
```
